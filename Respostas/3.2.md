```typescript
import { createActionGroup, emptyProps, props } from '@ngrx/store';
import { Todo } from './todo.model';

export const TodoActions = createActionGroup({
 source: 'Todo',
 events: {
   'Load Todos': emptyProps(),
   'Load Todos Success': props<{ todos: Todo[] }>(),
   'Load Todos Error': props<{ erro: string }>(),
   'Toggle Todo Complete': props<{ id: number }>(),
 }
});
```

```typescript
// State + Reducer
export interface TodoState {
 todos: Todo[];
 carregando: boolean;
 erro: string | null;
}

const estadoInicial: TodoState = {
 todos: [],
 carregando: false,
 erro: null,
};

export const todoFeature = createFeature({
 name: 'todo',
 reducer: createReducer(
   estadoInicial,

   on(TodoActions.loadTodos, state => ({
     ...state,
     carregando: true,
     erro: null,
   })),

   on(TodoActions.loadTodosSuccess, (state, { todos }) => ({
     ...state,
     todos,
     carregando: false,
   })),

   on(TodoActions.loadTodosError, (state, { erro }) => ({
     ...state,
     erro,
     carregando: false,
   })),

   on(TodoActions.toggleTodoComplete, (state, { id }) => ({
     ...state,
     todos: state.todos.map(todo =>
       todo.id === id ? { ...todo, concluido: !todo.concluido } : todo
     ),
   }))
 )
});
export const { selectTodoState, selectTodos, selectCarregando } = todoFeature;
```

```typescript
// Service
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { Todo } from './todo.model';


@Injectable({ providedIn: 'root' })
export class TodoService {
 private apiUrl = 'https://api.exemplo.com/todos';

 constructor(private http: HttpClient) {}

 buscarTodos(): Observable<Todo[]> {
   return this.http.get<Todo[]>(this.apiUrl);
 }
}
```

```typescript
// Effects
import { Injectable } from '@angular/core';
import { Actions, createEffect, ofType } from '@ngrx/effects';
import { catchError, map, switchMap } from 'rxjs/operators';
import { of } from 'rxjs';
import { TodoActions } from './todo.actions';
import { TodoService } from './todo.service';

@Injectable()
export class TodoEffects {
 constructor(
   private actions$: Actions,
   private todoService: TodoService
 ) {}

 loadTodos$ = createEffect(() =>
   this.actions$.pipe(
     ofType(TodoActions.loadTodos),      
     switchMap(() =>                      
       this.todoService.buscarTodos().pipe(
         map(todos => TodoActions.loadTodosSuccess({ todos })),
         catchError(err => of(TodoActions.loadTodosError({ erro: err.message })))
       )
     )
   )
 );
}
```

```typescript
// Configuração
import { provideStore } from '@ngrx/store';

   provideEffects(TodoEffects),
 ]
};
```
