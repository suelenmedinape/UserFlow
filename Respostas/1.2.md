
1. Eu comecei definindo as interfaces PaginaParams (para os dados de entrada) e Pagina<T> (para a resposta). Elas funcionam como os alicerces do código, garantindo que a paginação seja previsível e tipada, eliminando qualquer necessidade de usar o any.
2. A função filtrarEPaginar<T> foi desenhada para aceitar qualquer tipo de dado. O parâmetro filterFn recebe um predicado de filtro compatível com o método nativo .filter() do JavaScript.
3. como as interfaces de usuário costumam começar na página 1, eu usei a conta (pagina - 1) * tamanho para localizar o ponto de partida exato. Depois, o método .slice() entra em cena para extrair apenas aquela 'fatia' específica da nossa lista já filtrada.

```typescript
interface PaginaParams {
 pagina: number;
 tamanho: number;
}

interface Pagina<T> {
 items: T[];
 total: number;
}

function filtrarEPaginar<T>(
 data: T[],
 filterFn: (item: T) => boolean,
 params: PaginaParams,
): Pagina<T> {

 const dadosFiltrados = data.filter(filterFn);
 const total = dadosFiltrados.length;

 const inicio = (params.pagina - 1) * params.tamanho;
 const fim = inicio + params.tamanho;

 const items = dadosFiltrados.slice(inicio, fim);

 return { items, total };
}

interface Usuario {
 id: number;
 nome: string;
 ativo: boolean;
}

const usuarios: Usuario[] = [
 { id: 1, nome: 'Ana Silva', ativo: true },
 { id: 2, nome: 'Carlos Souza', ativo: true },
 { id: 3, nome: 'Ana Clara', ativo: false },
 { id: 4, nome: 'Beto Gomes', ativo: true },
 { id: 5, nome: 'Mariana Silva', ativo: true }
];

const termoBusca = 'Ana';
const parametros: PaginaParams = { pagina: 1, tamanho: 2 };

const resultado = filtrarEPaginar<Usuario>(
 usuarios,
 (usuario) => usuario.nome.toLowerCase().includes(termoBusca.toLowerCase()),
 parametros
);

console.log(resultado);
```
