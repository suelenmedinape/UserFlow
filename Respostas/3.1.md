
```typescript
import { Component, computed, signal, output, effect } from '@angular/core';

export interface ItemCarrinho {
 id: number;
 nome: string;
 preco: number;
 quantidade: number;
}

@Component({
 selector: 'app-carrinho',
 standalone: true,
 templateUrl: './carrinho.component.html'
})
export class CarrinhoComponent {

 itens = signal<ItemCarrinho[]>([]);

 total = computed(() =>
   this.itens().reduce((acc, item) => acc + item.preco * item.quantidade, 0)
 );

 totalMudou = output<number>();

 constructor() {
   effect(() => {
     this.totalMudou.emit(this.total());
   });
 }

 adicionarItem(novoItem: Omit<ItemCarrinho, 'quantidade'>): void {
   this.itens.update(lista => {
     const existente = lista.find(i => i.id === novoItem.id);
     if (existente) {
       return lista.map(i =>
         i.id === novoItem.id ? { ...i, quantidade: i.quantidade + 1 } : i
       );
     }
     return [...lista, { ...novoItem, quantidade: 1 }];
   });
 }

 removerItem(id: number): void {
   this.itens.update(lista =>
     lista
       .map(i => i.id === id ? { ...i, quantidade: i.quantidade - 1 } : i)
       .filter(i => i.quantidade > 0)
   );
 }
}
```

```html
<div>
 <h2>Carrinho</h2>

 @if (itens().length === 0) {
   <p>Seu carrinho está vazio.</p>
 } @else {
   @for (item of itens(); track item.id) {
     <div>
       <span>{{ item.nome }} — R$ {{ item.preco }} × {{ item.quantidade }}</span>
       <button (click)="adicionarItem(item)">+</button>
       <button (click)="removerItem(item.id)">-</button>
     </div>
   }
 }

 <strong>Total: R$ {{ total() | number: '1.2-2' }}</strong>
</div>
```
